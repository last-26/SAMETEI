# 1. ROTATION DETECTION - CV Tabanlı Otomatik Döndürme
def detect_rotation_cv(pil_img):
    """OpenCV ile yatay kenar analizi"""
    gray = cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2GRAY)
    
    # Sobel kenar tespiti - x ve y yönlerinde
    sobel_x = cv2.Sobel(gray, cv2.CV_32F, 1, 0, ksize=3)
    sobel_y = cv2.Sobel(gray, cv2.CV_32F, 0, 1, ksize=3)
    
    # Yatay/dikey kenar oranı
    score_orig = np.mean(np.abs(sobel_x)) / (np.mean(np.abs(sobel_y)) + 1e-6)
    
    # 90° sol ve sağ rotasyonları test et
    left_img = pil_img.rotate(90, expand=True)
    right_img = pil_img.rotate(-90, expand=True)
    
    # En iyi skoru bul (margin: 0.12)
    if score_left > score_orig + 0.12:
        return "left"  # 90° sola döndür
    elif score_right > score_orig + 0.12:
        return "right"  # 90° sağa döndür
    return "no"  # Döndürme gerekmez

# Dinamik token ayarlama
def calculate_tokens(image_size, strategy):
    width, height = image_size
    total_pixels = width * height
    
    if total_pixels > 1_000_000:  # Büyük görüntü
        if strategy == "table":
            # Tablo için daha fazla token ayır
            return min(2400, total_pixels // 400)
        else:
            return min(1800, total_pixels // 500)
    else:
        return 1200  # Standart limit

# Görüntü boyut optimizasyonu
def resize_for_ocr(image):
    """OCR için optimal boyuta getir"""
    MAX_PIXELS = 802816  # 896x896
    MIN_PIXELS = 501760  # 708x708
    
    current_pixels = image.width * image.height
    
    if current_pixels > MAX_PIXELS:
        # Küçült
        scale = (MAX_PIXELS / current_pixels) ** 0.5
        new_size = (int(image.width * scale), 
                   int(image.height * scale))
        return image.resize(new_size, Image.LANCZOS)
    
    return image

// Node.js ile OCR kullanımı
const LocalQwenVL = require('./utils/localQwenVL');

async function processHRDocument(imagePath) {
  const qwenVL = new LocalQwenVL('http://localhost:8000');
  
  // 1. Otomatik strateji belirleme
  const result = await qwenVL.extractFromImage(
    imagePath,
    'auto',  // Otomatik tip algılama
    null,    // Custom prompt yok
    {
      strategy: 'auto',
      output: 'json',      // JSON çıktı
      headers: ['Ad', 'Soyad', 'TC No', 'Departman']  // Tablo başlıkları
    }
  );
  
  if (result.success) {
    console.log('✅ OCR Başarılı!');
    console.log('📊 Strateji:', result.detectedStrategy);
    console.log('⏱️ Süre:', result.elapsedMs, 'ms');
    
    // JSON parse et
    const data = JSON.parse(result.text);
    
    // MongoDB'ye kaydet
    await saveToRAG({
      type: 'hr_document',
      content: data,
      extractedAt: new Date(),
      metadata: {
        originalFile: imagePath,
        ocrModel: 'Qwen2.5-VL-3B',
        processingTime: result.elapsedMs
      }
    });
    
    return data;
  }
}
